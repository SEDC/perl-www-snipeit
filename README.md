# NAME

WWW::SnipeIT - API Access to Snipe-IT

# SYNOPSIS

     use WWW::SnipeIT;

     my $snipeIT = SnipeIT->new( endpoint => 'http://<mysnipeip>/api/v1/', accessToken => 'mylongapikey');

     my $snipe = $snipeIT->snipe();
     my $result = $snipe->{'hardware'}->getHardwareIDByAssetTag (123);
     say $result;
